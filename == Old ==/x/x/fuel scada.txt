containerInvs = {}

function FluidCount(fluidIndex)
  count = 0
  for _,container in pairs(fluidIndex) do
      containerInvs[container] = container:getFluidType()
      count = container.fluidContent
     
      --if (count > 390) then
      --  container:flush()
      --end 
  end
  return count
end

function convertToPercentage(number)
    -- Ensure the number is within the valid range
    if number < 0 or number > 2400 then
        print("Input number must be between 0 and 2400")
    end

    -- Calculate the percentage
    local percentage = (number / 2400) * 100

    -- Return the calculated percentage
    return percentage
end

function ShowTankInfo(fluidTank, display, gauge)
	local fluidLevel = FluidCount(fluidTank)
	local fillPercent = convertToPercentage(fluidLevel)
	
	display:setText(string.format("%.0f", fluidLevel))
	gauge.percent = fillPercent / 100
end
